<script>
    const totalSteps = 100;
    let currentStep = 0;
    let quizHistory = []; 
    let isProcessing = false;
    let correctCount = 0;
    let quizStartTime = null;      
    let questionStartTime = null;  

    document.addEventListener('DOMContentLoaded', () => {
        quizStartTime = Date.now();
        loadNextQuiz();
    });

    function loadNextQuiz() {
        if (currentStep >= totalSteps) {
            showResults();
            return;
        }
        fetch('/quiz/api/get_data/')
            .then(res => res.json())
            .then(data => {
                if (data.error) return;
                renderQuiz(data);
            });
    }

    function renderQuiz(data) {
        currentStep++;
        const percent = (currentStep / totalSteps) * 100;
        document.getElementById('progress-bar').style.width = `${percent}%`;
        document.getElementById('question-text').innerText = data.target_def;

        const container = document.getElementById('options-container');
        container.innerHTML = '';
        container.classList.remove('quiz-locked');

        const shield = document.getElementById('click-shield');
        shield.style.pointerEvents = "none";

        data.options.forEach(opt => {
            const btn = document.createElement('div'); // 不再使用 <button>，讓只抓 button 的腳本失效

            // 為每個選項建立隨機 class / data 屬性，增加腳本選擇器難度
            const randKey = Math.random().toString(36).slice(2, 8);
            const randClass = `opt-${randKey}`;
            btn.className = `option-btn ${randClass} group relative bg-white border-4 border-black p-4 sm:p-6 rounded-[16px] sm:rounded-[24px] text-xl sm:text-2xl font-black shadow-[6px_6px_0px_0px_rgba(0,0,0,1)] hover:bg-sky-50 transition-all active:translate-y-1 active:shadow-none min-h-[52px] flex items-center justify-center break-words cursor-pointer`;
            btn.dataset.rand = randKey;
            btn.setAttribute('role', 'button');
            btn.setAttribute('tabindex', '0');

            // 額外包一層無語意 wrapper，讓 DOM 結構較難被固定腳本綁死
            btn.innerHTML = `
                <span class="wrap-a">
                    <span class="wrap-b">
                        <span class="relative z-10">${opt.word}</span>
                    </span>
                </span>
            `;

            // 注意：將 opt.id / 正確答案 ID 傳入後端驗證用
            const onSelect = () => handleAnswer(opt.id, data.correct_id, btn, data.target_word, data.target_def);
            btn.onclick = onSelect;
            btn.onkeydown = (e) => {
                if (e.key === 'Enter' || e.key === ' ') {
                    e.preventDefault();
                    onSelect();
                }
            };

            container.appendChild(btn);
        });

        setTimeout(() => {
            questionStartTime = Date.now(); 
            isProcessing = false; 
        }, 100); 
    }

    function handleAnswer(selectedId, correctId, btnElement, correctWord, def) {
        if (isProcessing) return; 
        isProcessing = true; 

        const endTime = Date.now();
        const startTime = questionStartTime || endTime;
        const timeSpent = parseFloat(((endTime - startTime) / 1000).toFixed(1));
        const isCorrect = String(selectedId) === String(correctId);
        
        const shield = document.getElementById('click-shield');
        shield.style.pointerEvents = "auto";
        
        if (isCorrect) {
            correctCount++; 
            btnElement.classList.add('btn-correct');
        } else {
            btnElement.classList.add('btn-wrong');
        }

        // --- 重點：儲存使用者選的內容與正確答案（含 selected_option 供後端驗證防刷分）---
        quizHistory.push({
            word_id: correctId,
            correct_word: correctWord,      // 正確答案
            user_answer: btnElement.innerText, // 使用者回答的文字
            selected_option: selectedId,    // 使用者點選的選項 ID，後端用來驗證是否真的答對
            isCorrect: isCorrect,
            timeSpent: Math.max(0.1, timeSpent),
            question_order: quizHistory.length + 1
        });

        setTimeout(() => {
            loadNextQuiz(); 
        }, 600);
    }

    function showResults() {
        const totalReactionTime = quizHistory.reduce((sum, item) => sum + item.timeSpent, 0).toFixed(1);
        const avgTime = (totalReactionTime / totalSteps).toFixed(1);
        const finalScore = Math.round((correctCount / totalSteps) * 100);

        document.getElementById('quiz-box').classList.add('hidden');
        document.getElementById('result-box').classList.remove('hidden');

        document.getElementById('res-total-time').innerText = totalReactionTime + 's';
        document.getElementById('res-avg-time').innerText = avgTime + 's';
        document.getElementById('res-score').innerText = `Score: ${finalScore}`;

        const tableBody = document.getElementById('review-table-body');
        // 確保 tableBody 所在的 table 標籤裡沒有 thead，或者直接清空
        // 這裡直接生成純資料行，並移除背景色干擾
        tableBody.innerHTML = quizHistory.map((item, index) => `
            <tr class="hover:bg-sky-50 transition-colors ${index === 0 ? '' : 'border-t-2 border-black'}">
                <td class="p-5">
                    <div class="flex items-center gap-4">
                        <span class="text-2xl ${item.isCorrect ? 'text-green-500' : 'text-red-500'} font-black">
                            ${item.isCorrect ? '✓' : '✗'}
                        </span>
                        <div>
                            <p class="text-[10px] font-black text-gray-400 uppercase leading-none mb-1">正確答案 Correct</p>
                            <p class="font-black text-2xl text-black">${item.correct_word}</p>
                        </div>
                    </div>
                </td>
                <td class="p-5">
                    <div>
                        <p class="text-[10px] font-black text-gray-400 uppercase leading-none mb-1">你的回答 Yours</p>
                        <p class="font-bold ${item.isCorrect ? 'text-green-600' : 'text-red-500'}">
                            ${item.isCorrect ? '正確' : item.user_answer}
                        </p>
                    </div>
                </td>
                <td class="p-5 text-right">
                    <p class="text-[10px] font-black text-gray-400 uppercase leading-none mb-1">用時</p>
                    <p class="font-black italic text-gray-400">${item.timeSpent}s</p>
                </td>
            </tr>
        `).join('');

        fetch('/quiz/api/submit/', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json', 'X-CSRFToken': 'kLR9PqNdYh1okOxmmRyvV7mNXHFBO7zqCJlPBUDj40EjqHtYun15X1MyIYB0iYef' },
            body: JSON.stringify({ 
                score: finalScore, 
                total_time: parseFloat(totalReactionTime), 
                correct_count: correctCount, 
                total_questions: totalSteps, 
                history: quizHistory 
            })
        })
        .then(res => res.json())
        .then(data => {
            if(data.status === 'success') {
                document.getElementById('res-exp').innerText = `+${data.exp_gained}`;
                document.getElementById('res-coin').innerText = `+${data.coin_gained}`;
            }
        });
    }

    function confirmExit() {
        if (confirm("確定要離開測驗嗎？進度將不會儲存。")) {
            window.location.href = "/quiz/select-mode/";
        }
    }
</script>

            if (modal) {
                modal.classList.add('hidden');
                modal.classList.remove('flex');
                var idsStr = modal.getAttribute('data-announcement-ids');
                if (idsStr) {
                    var ids = idsStr.split(',').map(function (x) { return parseInt(x, 10); }).filter(function (n) { return !isNaN(n); });
                    if (ids.length) {
                        var meta = document.querySelector('meta[name="csrf-token"]');
                        var token = meta ? meta.getAttribute('content') : '';
                        fetch('/api/announcements-mark-read/', {
                            method: 'POST',
                            headers: { 'Content-Type': 'application/json', 'X-CSRFToken': token },
                            body: JSON.stringify({ announcement_ids: ids })
                        }).then(function () {
                            var dot = document.getElementById('announcement-dot');
                            if (dot) dot.remove();
                        });
                    }
                }
            }
        }

                const icon = item.querySelector('.announcement-toggle-icon');

                if (!headerBtn || !content || !icon) return;

                headerBtn.addEventListener('click', function () {
                    const isOpen = !content.classList.contains('hidden');

                    // 先全部收合
                    items.forEach(function (it) {
                        const c = it.querySelector('.announcement-content');
                        const ic = it.querySelector('.announcement-toggle-icon');
                        if (c && ic) {
                            c.classList.add('hidden');
                            ic.textContent = '+';
                        }
                    });

                    // 如果原本是收合，現在展開這一個
                    if (!isOpen) {
                        content.classList.remove('hidden');
                        icon.textContent = '−';
                    }
                });
            });
        });
